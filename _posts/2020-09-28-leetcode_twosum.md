class Solution(object):
    def twoSum(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: List[int]
        """
        ids = [n for n in range(len(nums))]
        d = dict(zip(nums,ids))

        for i, num in enumerate(nums):
            diff = target - num
            try:
                other_id = d[diff]
                if i != other_id:
                    return (i, other_id)
            except:
                pass
