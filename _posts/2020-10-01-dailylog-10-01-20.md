## Naive solution

```python
class RecentCounter(object):

    def __init__(self):
        self.results = []


    def ping(self, t):
        """
        :type t: int
        :rtype: int
        """
        print([t-3000, t])

        curr_range = [t-3000, t]
        self.results.append(t)
        temp = []
        for r in self.results:
            if curr_range[0] <= r <= curr_range[1]:
                temp.append(r)

        return len(temp)
```
